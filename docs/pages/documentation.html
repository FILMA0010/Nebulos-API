Example

Some examples how to use the code within a website 

            
<!DOCTYPE html>
<html>
<head>
    <title>API Example</title>
    <script src="https://code.jquery.com/jquery-3.6.0.min.js"></script>
</head>
<body>
    <h1>JSON API Example</h1>
    <div id="api-response"></div>

    <script>
        $(document).ready(function() {
            $.ajax({
                url: "https://api.nebulos.pro:5000/api/text/testyourapi",
                type: "GET",
                dataType: "json",
                success: function(response) {
                    var message = response.message;
                    $("#api-response").text("Output: " + message);
                },
                error: function(error) {
                    console.log(error);
                    $("#api-response").text("Error by using Nebulos API");
                }
            });
        });
    </script>
</body>
</html>
